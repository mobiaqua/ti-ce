    43    
    44    metaonly module EDMA3CHAN inherits ti.sdo.fc.ires.IRESMAN {
    45    
    46        /*  @_nodoc */
    47        override config Bool autoRegister = true;
    48    
    49        /*!
    50         *  ======== ipcKeyBase ========
    51         *  Linux only. Base value of keys of Linux IPC objects used by
    52         *  EDMA3CHAN. The IPC objects created by EDMA3CHAN will use keys starting
    53         *  at this
    54         *  value, and incrementing with each new object. There are currently
    55         *  three IPC objects, so keys in the range of ipcKeyBase to ipcKeyBase + 2
    56         *  will be reserved for EDMA3CHAN. The default value of ipcKeyBase is
    57         *  ascii code for "NACH".
    58         *
    59         *  WARNING: This value should only be changed if it conflicts with
    60         *  another IPC key in the system that cannot be changed. If this value
    61         *  is changed, all programs using EDMA3CHAN that will be run simultaneously
    62         *  must have the ipcKeyBase configured to the new value.
    63         *  @_nodoc
    64         */
    65        config UInt ipcKeyBase = 0x4e414843;
    66    
    67    }
