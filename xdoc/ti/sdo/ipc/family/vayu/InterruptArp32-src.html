    37    import ti.sdo.utils.MultiProc;
    38    
    39    /*!
    40     *  ======== InterruptArp32 ======== 
    41     *  ARP32 based interrupt manager
    42     */
    43    
    44    module InterruptArp32 inherits ti.sdo.ipc.notifyDrivers.IInterrupt
    45    {
    46        /* Total number of cores on Vayu SoC */
    47        const UInt8 NUM_CORES = 7;
    48    
    49        /* Number of Cores in EVE Sub-system */
    50        const UInt8 NUM_EVES = 4;
    51    
    52        /* Number of Internal EVE mailboxes */
    53        const UInt8 NUM_EVE_MBX = 12;
    54    
    55        /* Number of System Mailboxes */
    56        const UInt8 NUM_SYS_MBX = 1;
    57    
    58        /* Base address for the Mailbox subsystem */
    59        config UInt32 mailboxBaseAddr[NUM_EVE_MBX + NUM_SYS_MBX];
    60    
    61        /*
    62         * Mailbox table for storing encoded Base Address, mailbox user Id,
    63         * and sub-mailbox index.
    64         */
    65        config UInt32 mailboxTable[NUM_CORES * NUM_CORES];
    66    
    67        config UInt32 eveInterruptTable[NUM_CORES];
    68    
    69        config UInt32 procIdTable[NUM_CORES];
    70    
    71    internal:
    72    
    73        /*! Statically retrieve procIds to avoid doing this at runtime */    
    74        config UInt eve0ProcId     = MultiProc.INVALIDID;
    75        config UInt eve1ProcId     = MultiProc.INVALIDID;
    76        config UInt eve2ProcId     = MultiProc.INVALIDID;
    77        config UInt eve3ProcId     = MultiProc.INVALIDID;
    78        config UInt dsp0ProcId     = MultiProc.INVALIDID;
    79        config UInt dsp1ProcId     = MultiProc.INVALIDID;
    80        config UInt benelliProcId  = MultiProc.INVALIDID;
    81    
    82        /*! Function table */
    83        struct FxnTable {
    84            Fxn    func;
    85            UArg   arg;
    86        }
    87    
    88        /*! Stub to be plugged for dsp-arp32 interrupts */
    89        Void intShmStub(UArg arg);
    90        
    91        struct Module_State {        
    92            /* 
    93             * Create a function table of length 8 (Total number of cores in the
    94             * System) for each EVE core.
    95             */
    96            FxnTable   fxnTable[NUM_CORES];
    97    
    98            /* 
    99             * numPlugged is used to track number of times the interrupt was
   100             * registered.
   101             *
   102             * numPlugged array length equals number of internal mailboxes per
   103             * EVE.
   104             */
   105            UInt numPlugged[NUM_EVE_MBX/NUM_EVES];
   106        };
   107    }
