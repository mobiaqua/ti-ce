    38    package ti.sdo.ce.views;
    39    import xdc.rov.ViewInfo;
    40    
    41    module CodecEngine {
    42    
    43            struct NODE_Uuid {
    44            UInt32 data;
    45            };
    46    
    47            enum Engine_CachedMemType {
    48                Engine_USECACHEDMEM_DEFAULT = -1,  /**< Use default cache setting */
    49                Engine_USECACHEDMEM_NONCACHED = 0, /**< Use non-cached memory */
    50                Engine_USECACHEDMEM_CACHED = 1     /**< Use cached memory */
    51            };
    52    
    53            struct Engine_AlgDesc {
    54                String      name;           /**< Name of algorithm. */
    55                NODE_Uuid   uuid;           /**< Fixed length perfect hash of name. */
    56                Ptr         fxns;          /**< If null, served by RMS. */
    57                Ptr         idmaFxns;       /**< DMA resource negotiation when not null. */
    58                String      *typeTab;       /**< inheritance hierarchy. */
    59                Bool        isLocal;        /**< if TRUE, run locally. */
    60                Int         groupId;        /**< Resource sharing group Id. */
    61                Int         rpcProtocolVersion; /**< Protocol version. */
    62                Ptr         iresFxns;       /**< IRES_Fxns when not null. */
    63                Ptr        codecClassConfig; /**< codec class config data, if any */
    64                Engine_CachedMemType memType;  /**< Memory type for alg's mem reqs. */
    65            };
    66    
    67            struct Engine_Desc {
    68                String          name;       /**< Name of this Engine. */
    69                Engine_AlgDesc  *algTab;    /**< Table of algs available in this Engine. */
    70                String          remoteName; /**< Optional name of Server image to load. */
    71                String          linkCfg;    /**< Optional Link config (when using LAD). */
    72                Int             numAlgs;    /**< Number of algorithms in algTab */
    73                Int             heapId;     /**< Heap ID for engine allocs. */
    74            };
    75    
    76            struct Engine_Config {
    77                Engine_Desc *engineTab;
    78                String      localEngine;
    79            };
    80    
    81            metaonly struct Engine_Meta {
    82            String algName;
    83            Bool   isLocal;
    84            Int    heapId;
    85        };
    86    
    87        @Facet
    88        metaonly config ViewInfo.Instance rovViewInfo =
    89            ViewInfo.create({
    90                viewMap: [
    91                    ['Basic', {type: ViewInfo.MODULE_DATA, viewInitFxn: 'viewInitBasic', structName: 'Engine_Meta'}],
    92                ]
    93            });
    94    
    95    
    96    }
